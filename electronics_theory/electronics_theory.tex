\section*{Force}

\begin{equation}
N_{force} = k_e\frac{Q_{1} \cdot Q_{2}}{L^2_{distance}}
\end{equation}

\section*{Energy}

\begin{equation}
J_{energy} = W_{power} \cdot T_{second}
\end{equation}

\begin{equation}
J_{energy} = N_{force} \cdot L_{distance}
\end{equation}

\begin{equation}
J_{energy} = Q_{charge} \cdot V_{voltage}
\end{equation}

\begin{equation}
J_{energy} = A^2_{current} \cdot \Omega \cdot T_{second}
\end{equation}

\section*{Power}

\begin{equation}
W_{power} = V_{voltage} \cdot A_{current}
\end{equation}

\begin{equation}
W_{power} = A^2_{current} \cdot \Omega_{resistance}
\end{equation}

\begin{equation}
W_{power} = \frac{J_{energy}}{T_{second}}
\end{equation}

\begin{equation}
W_{power} = \frac{V^2_{voltage}}{\Omega_{resistance}}
\end{equation}

\section*{Electric charge}

\begin{equation}
Q_{charge} = A_{current} \cdot T_{second}
\end{equation}

\begin{equation}
Q_{charge} = \frac{J_{energy}}{V_{voltage}}
\end{equation}

\begin{equation}
Q_{charge} = n_{electrones} \cdot e_{elementary\_charge}
\end{equation}

\begin{equation}
1Q_{charge} = 1A_{current} \cdot T_{second}
\end{equation}

\section*{Electric field}

\begin{equation}
E_{electric\_field} = k_e\frac{Q_{charge}}{L^2_{distance}}
\end{equation}

\section*{Current}

\begin{equation}
A_{current} = \frac{Q_{charge}}{T_{second}}
\end{equation}

\section*{Electric potential}

\begin{equation}
V_{voltage} = \frac{J_{energy}}{Q_{charge}}
\end{equation}

\section*{Condensator}

\begin{equation}
J_{energy} = \frac{1}{2} \cdot C_{capacity} \cdot V^2_{voltage}
\end{equation}
